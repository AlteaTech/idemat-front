CREATE TRIGGER [dbo].[TRG_AFTERUPDATE_ACHATPASSAGES]
   ON  [dbo].[TB_WA_ACHATPASSAGES] 
   AFTER INSERT, UPDATE
AS 
BEGIN
	SET NOCOUNT ON;
	
	IF UPDATE(STR_PAYFIP_RESULTRANS)
	BEGIN
		DECLARE @ancienne char(1)
		DECLARE @nouvelle char(1)
		SELECT @ancienne = STR_PAYFIP_RESULTRANS FROM DELETED
		SELECT @nouvelle = STR_PAYFIP_RESULTRANS FROM INSERTED

		IF (@ancienne IS NULL AND @nouvelle IS NOT NULL) OR @ancienne <> @nouvelle
		BEGIN
			IF @nouvelle = 'P'
			BEGIN
				INSERT INTO TB_ACHATPASSAGES (LNG_GUIDUSAGER, STR_DATEACHAT, INT_NBPASSAGES) SELECT LNG_GUIDUSAGER, STR_DATEACHAT, INT_NBPASSAGES FROM INSERTED
			END
		END	
	END
END
